# Math
* +, -, *, /, % **
```python
print("5 + 2 =", 5 + 2) # 7
print("5 - 2 =", 5 - 2) # 3
print("5 * 2 =", 5 * 2) # 10
print("5 / 2 =", 5 / 2) # 2.5
print("5 % 2 =", 5 % 2) # 1
print("5 ** 2 =", 5 ** 2) # 25
print("5 // 2 =", 5 // 2) # 2

i1 = 2
i1 += 1 # 3
```
